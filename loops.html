游대 Loops

While Loops

# Basic while loop count = 0 while count < 5: print(count) count += 1 # Output: 0, 1, 2, 3, 4

For Loops

# Loop with range for i in range(5): print(i) # 0, 1, 2, 3, 4 # Loop with start and stop for i in range(2, 6): print(i) # 2, 3, 4, 5 # Loop with step for i in range(0, 10, 2): print(i) # 0, 2, 4, 6, 8 # Reverse loop for i in range(10, 0, -1): print(i) # 10, 9, 8, ..., 1 # Using reversed() for i in reversed(range(1, 6)): print(i) # 5, 4, 3, 2, 1

Nested Loops

# Print all pairs for i in range(1, 4): for j in range(1, 4): print(f"({i}, {j})") # Output: (1,1), (1,2), (1,3), (2,1), (2,2), ...

Loop Control

# break - exit loop for i in range(10): if i == 5: break print(i) # 0, 1, 2, 3, 4 # continue - skip iteration for i in range(5): if i == 2: continue print(i) # 0, 1, 3, 4 # pass - do nothing for i in range(5): pass # Placeholder
