C# vs Python pattern translation (side-by-side)

A) Frequency count

Python

freq = {} for x in arr: freq[x] = freq.get(x, 0) + 1

C#

var freq = new Dictionary<int,int>(); foreach (var x in arr) { freq[x] = freq.GetValueOrDefault(x, 0) + 1; }

B) Two pointers

Python

l, r = 0, len(arr) - 1 while l < r: l += 1 r -= 1

C#

int l = 0, r = arr.Length - 1; while (l < r) { l++; r--; }

C) Sliding window (grow/shrink)

Python

start = 0 for end in range(len(nums)): # expand using nums[end] while condition_not_met: start += 1

C#

int start = 0; for (int end = 0; end < nums.Length; end++) { // expand using nums[end] while (conditionNotMet) { start++; } }

D) Stack (valid parentheses)

Python

pairs = {')':'(', ']':'[', '}':'{'} stack = [] for ch in s: if ch in "([{": stack.append(ch) else: if not stack or stack[-1] != pairs[ch]: return False stack.pop() return not stack

C#

var pairs = new Dictionary<char,char> { {')','('}, {']','['}, {'}','{'} }; var st = new Stack<char>(); foreach (var ch in s) { if (ch == '(' || ch == '[' || ch == '{') st.Push(ch); else { if (st.Count == 0 || st.Peek() != pairs[ch]) return false; st.Pop(); } } return st.Count == 0;

E) BFS (queue)

Python

from collections import deque q = deque([start]) seen = {start} while q: node = q.popleft() for nei in graph[node]: if nei not in seen: seen.add(nei) q.append(nei)

C#

var q = new Queue<int>(); var seen = new HashSet<int>(); q.Enqueue(start); seen.Add(start); while (q.Count > 0) { int node = q.Dequeue(); foreach (var nei in graph[node]) { if (seen.Add(nei)) q.Enqueue(nei); } }

F) Binary search

Python

l, r = 0, len(arr) - 1 while l <= r: m = (l + r) // 2 if arr[m] == target: return m elif arr[m] < target: l = m + 1 else: r = m - 1

C#

int l = 0, r = arr.Length - 1; while (l <= r) { int m = l + (r - l) / 2; if (arr[m] == target) return m; if (arr[m] < target) l = m + 1; else r = m - 1; } return -1;
